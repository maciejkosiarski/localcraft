{{/* credentials-list.html Reusable component for displaying certificates, diplomas, awards. Usage: {{ partial "credentials-list.html" (dict "items" $credentials "limit" 3 "id" "team-anna") }} Parameters: - items: slice of credential objects (name, issuer, date, image?, icon?, description?) - limit: number of items to show before toggle (default: 6) - id: unique identifier for toggle functionality */}} {{- $items := .items -}} {{- $limit := .limit | default 6 -}} {{- $id := .id | default "credentials" -}} {{- if $items -}} {{/* Sort by date descending (newest first) */}} {{- $sorted := sort $items "date" "desc" -}} {{- $total := len $sorted -}} {{- $visible := first $limit $sorted -}} {{- $hidden := after $limit $sorted -}} {{- $hasMore := gt $total $limit -}}
{{/* Visible items */}}
{{- range $visible -}} {{- partial "credentials-item.html" . -}} {{- end -}}
{{/* Hidden items (if any) */}} {{- if $hasMore -}}
{{- range $hidden -}} {{- partial "credentials-item.html" . -}} {{- end -}}
{{- end -}}
{{- end -}}
